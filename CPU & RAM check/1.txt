      Write-host "Raspberry Pi system is perferctly normal."
      Do
         {
     		$RAM= (Get-counter '\Memory\Available Mbytes').CounterSamples.Cookedvalue

     		write-host "The total RAM left is:"$RAM
     		write-host "The total CPU consumption is:"$CPU
         }
      until (1 -gt 2)
}
